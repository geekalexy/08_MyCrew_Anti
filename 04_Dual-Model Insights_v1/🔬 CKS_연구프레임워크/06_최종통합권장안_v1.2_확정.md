### 🆕 [v1.2 추가] KSI-S 구현 코드 — 현행 + 다음 단계 병행

#### 🟢 현행 구현 (실시간 루프용 — 속도·재현성 우선)

```python
from sentence_transformers import SentenceTransformer, util
import sqlite3

# 모델: MiniLM-L12-v2 (경량, 한국어 지원, 로컬 구동)
model = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')

def calculate_ksi_s(A_pre_text, B_pre_text, A_post_text, B_post_text):
    """
    초기 산출물(pre)과 회고 이후 산출물(post)로
    의미론적 지식 동기화 지수(KSI-S)를 계산.
    """
    vec_A_pre  = model.encode(A_pre_text)
    vec_B_pre  = model.encode(B_pre_text)
    vec_A_post = model.encode(A_post_text)
    vec_B_post = model.encode(B_post_text)

    # 베이스라인: Phase 1에서 A와 B가 원래 얼마나 비슷했나?
    cs_baseline      = util.cos_sim(vec_A_pre, vec_B_pre).item()

    # 교차 흡수율
    cs_A_absorbed_B  = util.cos_sim(vec_A_post, vec_B_pre).item()
    cs_B_absorbed_A  = util.cos_sim(vec_B_post, vec_A_pre).item()

    # 델타 (순수하게 교차 평가로 증가한 동기화 값)
    ksi_delta_A = cs_A_absorbed_B - cs_baseline
    ksi_delta_B = cs_B_absorbed_A - cs_baseline
    ksi_s_final = (ksi_delta_A + ksi_delta_B) / 2

    return ksi_s_final

# AutoGen Message Loop Hook에서 호출 → SQLite 자동 적재
# db.execute("INSERT INTO ksi_metrics VALUES (?, ?)", (task_id, ksi_s_final))
```

#### 🔵 다음 단계 구현 (논문 최종 데이터용 — 정밀도 우선, 교차 검증)

```python
# [v1.2 다음 단계] mpnet 모델 병행 로드 — 논문 데이터 정밀 추출 전용
# paraphrase-multilingual-mpnet-base-v2: MiniLM 대비 정밀도 높으나 속도 낮음
model_paper = SentenceTransformer('paraphrase-multilingual-mpnet-base-v2')

def calculate_ksi_s_paper(A_pre_text, B_pre_text, A_post_text, B_post_text):
    """
    논문 최종 데이터 추출 전용 고정밀 버전.
    MiniLM 결과와 비교하여 모델 간 일치 여부를 검증한다.
    """
    # MiniLM과 동일 로직, 모델만 교체
    vec_A_pre  = model_paper.encode(A_pre_text)
    vec_B_pre  = model_paper.encode(B_pre_text)
    vec_A_post = model_paper.encode(A_post_text)
    vec_B_post = model_paper.encode(B_post_text)

    cs_baseline     = util.cos_sim(vec_A_pre, vec_B_pre).item()
    cs_A_absorbed_B = util.cos_sim(vec_A_post, vec_B_pre).item()
    cs_B_absorbed_A = util.cos_sim(vec_B_post, vec_A_pre).item()

    ksi_s_final = ((cs_A_absorbed_B - cs_baseline) +
                   (cs_B_absorbed_A - cs_baseline)) / 2
    return ksi_s_final
```
